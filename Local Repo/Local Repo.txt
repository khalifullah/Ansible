    - name: Create APPSTREAM repository
      yum_repository:
        name: APPSTREAM
        description: "APPstream software"
        baseurl: http://server.network.example.com/AppStream
        enabled: true

        gpgkey: http://server.network.example.com/RHEL/RPM-GPG-KEY-redhat-release
